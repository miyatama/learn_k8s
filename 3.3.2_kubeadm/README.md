# outline

use GCP VM instance.

## installation

master and nodes setting.see [kubectl installation](https://kubernetes.io/ja/docs/tasks/tools/install-kubectl/).

```shell
sudo apt update && sudo apt install -y apt-transport-https curl

curl -s https://packages.cloud.google.com/apt/doc/apt-key.gpg | sudo apt-key add - $ cat << EOF | sudo tee /etc/apt/sources.list.d/kubernetes.list
deb https://apt.kubernetes.io/kubernetes-xenial main
EOF
sudo apt update
sudo apt-get -y install kubelet=1.18.2-00 kubeadm=1.18.2-00 kubectl=1.18.2-00 docker.io
# immutable version 
sudo apt-mark hold kubelet kubeadm kubectl docker.io
# create overlay network
cat << EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
net.ipv4.ip_forward = 1
EOF
sudo sysctl --system
# disable swap.eternal setting is comment out swap setting in /etc/fstab.
sudo swapoff -a
```

need match server version

```shell
sudo apt-cache policy kubectl
# sudo apt policy kubectl
# sudo apt list kubectl -a
sudo apt-cache showpkg kubectl
sudo apt-get -y install kubectl=1.18.0-00
```

setup kubeadm.see [kubeadm installation](https://kubernetes.io/ja/docs/setup/production-environment/tools/kubeadm/install-kubeadm/).

```shell
#!/bin/bash

# update package
sudo apt update -y
sudo apt upgrade -y
# install tools
sudo apt install -y \
  apt-transport-https \
  curl \
  ca-certificates \
  gnupg \
  lsb-release

# install docker.see https://docs.docker.com/engine/install/debian/
curl -fsSL https://download.docker.com/linux/debian/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
## use stable
echo \
  "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/debian \
  $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
sudo apt-get update -y
sudo apt-get install -y \
  docker-ce \
  docker-ce-cli \
  containerd.io
# install kubectl
curl -s https://packages.cloud.google.com/apt/doc/apt-key.gpg | sudo apt-key add -
echo "deb https://apt.kubernetes.io/ kubernetes-xenial main" | sudo tee -a /etc/apt/sources.list.d/kubernetes.list
sudo apt-get update
## check version
## apt-cache showpkg kubectl
sudo apt-get install -y \
  kubelet=1.22.5-00 \
  kubeadm=1.22.5-00 \
  kubectl=1.22.5-00
## immutable version 
sudo apt-mark hold kubelet kubeadm kubectl
# create overlay network
cat << EOF | sudo tee /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-ip6tables = 1
net.bridge.bridge-nf-call-iptables = 1
net.ipv4.ip_forward = 1
EOF
sudo sysctl --system
# disable swap.eternal setting is comment out swap setting in /etc/fstab.
sudo swapoff -a
## check swap
## sudo swapon -s
# setup iptables
sudo apt-get install -y iptables arptables ebtables
## change legacy version
sudo update-alternatives --set iptables /usr/sbin/iptables-legacy
sudo update-alternatives --set ip6tables /usr/sbin/ip6tables-legacy
sudo update-alternatives --set arptables /usr/sbin/arptables-legacy
sudo update-alternatives --set ebtables /usr/sbin/ebtables-legacy
# docker daemon setting
sudo touch /etc/docker/daemon.json
cat << EOF | sudo tee -a /etc/docker/daemon.json
{
    "exec-opts": ["native.cgroupdriver=systemd"]
}
EOF
```

and reboot

```shell
sudo reboot
```
